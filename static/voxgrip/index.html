Apps Script Guide

  1. Create a new Google Sheet.
  2. Go to Extensions > Apps Script.
  3. Paste the code block below into the script editor.
  4. Click Deploy > New Deployment.
  5. Set "Execute as" to Me and "Who has access" to Anyone.
  6. Deploy, authorize permissions, copy the Web App URL, and paste it here!

function doPost(e) {
  var sheet = SpreadsheetApp.getActiveSpreadsheet().getActiveSheet();
  var data = JSON.parse(e.postData.contents);
  
  // Verify secret token
  var secretToken = "voxgrip_secret_12345";
  if (data.token !== secretToken) {
    return ContentService.createTextOutput(JSON.stringify({status: "error", message: "Unauthorized"})).setMimeType(ContentService.MimeType.JSON);
  }
  
  // Add headers if sheet is empty
  if (sheet.getLastRow() === 0) {
    sheet.appendRow([
      "Timestamp", "Session ID", "Total Dur (s)", "Voiced Dur (s)", 
      "F0 Pitch (Hz)", "Jitter (%)", "Shimmer (%)", 
      "F1 Formant (Hz)", "F2 Formant (Hz)", "Vowel Space", 
      "Back-Pressure Index", "Dryness/Friction Index", "Remedials / Recommendations"
    ]);
  }
  
  // Append data row
  sheet.appendRow([
    data.timestamp,
    data.session_id,
    data.total_duration_sec,
    data.voiced_duration_sec,
    data.f0_hz,
    data.jitter_percentage,
    data.shimmer_percentage,
    data.f1_formant_hz,
    data.f2_formant_hz,
    data.vowel_space,
    data.back_pressure_index,
    data.dryness_friction_index,
    data.remedials
  ]);
  
  return ContentService.createTextOutput(JSON.stringify({status: "success"})).setMimeType(ContentService.MimeType.JSON);
}
